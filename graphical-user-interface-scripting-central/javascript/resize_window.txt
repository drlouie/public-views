<script language="JavaScript" >

function resize() {

	if (isOpen==0) {
		Open(WidthMax)
	} else {
		Close(WidthMin);
	}
}

function Open(Width) {
	if (browserName == "Netscape") {
		if (Width < MaxWidth) {
			parent.window.resizeTo(Width,fixedHeight);
			xWidth=parent.window.outerWidth
			xWidth = xWidth + 15;
			setTimeout('Open(xWidth)',0);
		} else {
			parent.window.resizeTo(MaxWidth,fixedHeight);
			isOpen=1;
			setTimeout('isOpen=1;',50);
		 }
	}else {
		if (Width < MaxWidth) {
			parent.window.resizeTo(Width,fixedHeight);
			xWidth=Width+15;
			setTimeout('Open(xWidth)',0);
		} else {
			parent.window.resizeTo(MaxWidth,fixedHeight);
			isOpen=1;
			setTimeout('isOpen=1;',50);
		 }
		
	}	 
}

function Close(Width) {
	if (browserName == "Netscape") {
		if (Width > MinWidth) {
			parent.window.resizeTo(Width,fixedHeight);
			xWidth=parent.window.outerWidth;
			xWidth=Width - 15;
			setTimeout('Close(xWidth)',0);
		} else {
			parent.window.resizeTo(MinWidth,fixedHeight);
			isOpen=0;
			setTimeout('isOpen=0;',50);
		} 
	}else {
			if (Width > MinWidth) {
			parent.window.resizeTo(Width,fixedHeight);
			xWidth=Width-15;
			setTimeout('Close(xWidth)',0);
		} else {
			parent.window.resizeTo(MinWidth,fixedHeight);
			isOpen=0;
			setTimeout('isOpen=0;',50);
		} 
	
	}	
}
